import math
def b(n):
    if n == 0:
        return 1
    elif n > 0:
        return sum([(-1)**(k+1) * math.comb(n, k) * 2**(k * (n - k)) * b(n - k) for k in range(1, n+1)])
    
b(3), b(10)
